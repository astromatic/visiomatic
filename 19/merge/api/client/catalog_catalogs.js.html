		const	prop = feature.properties.items,
			djd =  (this.jd[1] - this.jd[0]) * 24.0,

			invclat = clat > 0. ? 1. / clat : 0.001,
			dlng = invclat * djd * prop[3] / 7200.,

		// Check to see if the delimiter is defined. If not, then default to comma.
		const	badreg = /#|No\s|^$/,
			sexare = /^([-+]?)(\d+)\s(\d+)\s(\d+\.?\d*)/,
